Tokens de movimento

Variáveis oficiais de easing e duração
--ease-* / motion-* 
/* ═══ GRUPO ALT — MOTION TOKENS ═══ */
:root {
  /* Easing curves */
  --ease-out:    cubic-bezier(0.22, 1, 0.36, 1);
  --ease-spring: cubic-bezier(0.34, 1.2, 0.64, 1);
  --ease-smooth: cubic-bezier(0.16, 1, 0.3, 1);

  /* Durations */
  --motion-instant: 100ms;  /* feedback visual */
  --motion-fast:    200ms;  /* hover, focus */
  --motion-base:    400ms;  /* transições padrão */
  --motion-slow:    900ms;  /* surgimento hero */
  --motion-ambient: 12s;    /* background orbs */
}
